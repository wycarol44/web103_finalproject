enums
submission_vote         UP, DOWN
case_phase              PROVISIONAL, DISCOVERY, ARGUMENT, JURY_DELIBERATION, RULING, CLOSED
juror_vote              GUILTY, NOT_GUILTY, INSUFFICIENT_EVIDENCE
verdict                 GUILTY, NOT_GUILTY, TB_PECKED_AT
xp_event_reason         SUBMISSION, JURY_VOTE, CASE_ADVANCED_TO_TRIAL, SUBMISSION_VOTE, SUBMISSION_CITED
xp_event_ref_type       CASE, EVIDENCE, ARGUMENT, VOTE

cases
---------
case_id             PK
user_id             FK -> users         -- submitted by
created_at          TIMESTAMPTZ         -- submission time
accusation          VARCHAR(200)
image_url           VARCHAR(500)
verdict             ENUM                -- GUILTY, NOT_GUILTY, TB_PECKED_AT (if votes don't meet threshold)
judge_id            FK -> users
judge_ruling        VARCHAR(300)        -- object designation based on verdict
phase               ENUM                -- PROVISIONAL, DISCOVERY, ARGUMENT, JURY_DELIBERATION, RULING, CLOSED
phase_start         TIMESTAMPTZ
phase_end           TIMESTAMPTZ

evidence
---------
evidence_id         PK
case_id             FK -> cases
user_id             FK -> users
evidence_num        INT                 -- relative to case, so they can be cited easier in arguments, generated on backend
content             VARCHAR(150)
image_url           VARCHAR(500)
created_at          TIMESTAMPTZ         -- submission time
UNIQUE(case_id, evidence_num)

arguments
---------
arg_id              PK
case_id             FK -> cases
user_id             FK -> users
arg_num             INT                 -- relative to case, so they can be cited easier by jurors 
text                VARCHAR(600)
created_at          TIMESTAMPTZ         -- submission time
UNIQUE(case_id, arg_num)

argument_case_refs
---------
arg_id              FK -> arguments
case_id             FK -> cases         -- precendent verdict referenced in argument
PRIMARY KEY(arg_id, case_id)

argument_evidence_refs
---------
arg_id              FK -> arguments
evidence_id         FK -> evidence      -- evidence from the current case referenced in argument
PRIMARY KEY(arg_id, evidence_id)

case_submission_votes
---------
case_id             FK -> cases
user_id             FK -> users
vote                ENUM                -- UP, DOWN                
PRIMARY KEY(case_id, user_id)

evidence_votes
---------
evidence_id         FK -> evidence
user_id             FK -> users
vote                ENUM                -- UP, DOWN
PRIMARY KEY(evidence_id, user_id)

argument_votes
---------
arg_id              FK -> arguments
user_id             FK -> users
vote                ENUM                -- UP, DOWN
PRIMARY KEY(arg_id, user_id)


jury_arg_refs
---------
jv_id               FK -> jury_assignments(id)
arg_id              FK -> arguments      -- can select multiple args as most convincing
PRIMARY KEY(jv_id, arg_id)

users
---------
user_id             PK
username            VARCHAR(40)
image_url           VARCHAR(500)
bio                 VARCHAR(500)
total_xp            INT                 -- cached value
created_at          TIMESTAMPTZ         -- submission time
UNIQUE(username)

jury_assignments
---------
id                  PK
case_id             FK -> cases
user_id             FK -> users
vote                ENUM                -- GUILTY, NOT_GUILTY, INSUFFICIENT_EVIDENCE
created_at          TIMESTAMPTZ         -- submission time
completed_at        TIMESTAMPTZ         -- submission time
UNIQUE(case_id, user_id)

xp_events
---------
xp_event_id         PK
user_id             FK -> users
amount              INT
reason              ENUM                -- SUBMISSION, JURY_VOTE, CASE_ADVANCED_TO_TRIAL, SUBMISSION_VOTE...
reference_type      ENUM                -- CASE, EVIDENCE, ARGUMENT, VOTE
reference_id        BIGINT              -- ID of the object that caused the XP
created_at          TIMESTAMPTZ         -- submission time
UNIQUE(user_id, reason, reference_type, reference_id)

achievements
---------
achievement_id      PK
name                VARCHAR(80)
requirements        VARCHAR(200)        -- description for users

user_achievements
---------
user_id             FK -> users
achievement_id      FK -> achievements
progress            INT                 -- for achievements that require user do action N times
threshold           INT
earned_at           TIMESTAMPTZ
highlighted         BOOL
PRIMARY KEY(user_id, achievement_id)
